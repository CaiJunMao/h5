字符串

			注意：JavaScript中字符串是不可变的。任何涉及到字符串变化的操作都不会修改源字符串，而是创建了一个新的字符串
			字符串类型
				 * 使用new关键字获取的是引用类型(对象类型为String)的字符串，
				 * var s = new String("abc");
					alert(typeof s);  //弹出：object ,这是一个String,即字符串对象		
				 * 不使用new的是基本类型的字符串,包括字面量和String()强转得到的
				 * var s = "good";	// s为基本类型的字符串
					alert(typeof s);  //弹出: string
			字符串常用方法：
				 * 字符串常用方法
				 * 【s.charAt(index)】返回的是指定位置的字符 == 直接用str[下标]
					 * Javascript没有字符类型，其实返回的是长度为 1 的字符串。
					 * var str = "你好呀"
					 * console.log(str.charAt(0))//你
					 * str.chatAt()//你，因为默认从0开始，但是不提倡
					 * str[0]//你，但是其实这是个对象
					 可以把字符串想象成一个数组，下标从0开始，也有length属性，最后一个元素的下标是：s.length - 1，
					 但是不能直接给str[0]="赋值"这样子给字符串插入或修改
				 * s.charCodeAt(index) 返回指定位置的字符的 Unicode 编码
					 * 这个返回值是 0 - 65535 之间的整数
					 * 常见ASCII码
					 * 0-48
					 * 9-57
					 * A-65
					 * Z-90
					 * a-97
					 * z-122
				 * 字符串连接s.concat(stringX,stringX,...,stringX) ，直接+更常用
					 * 如果连接一个长度为 0的字符串，比如“”那么返回的是字符串本身
					 * 并不会改变源字符串
					 * 实际开发中，使用 字符串连接符( + )，更简洁
					 * console.log("我在粤嵌"+"学习")
				 * 查找子字符串
					 * s.indexOf(searchvalue,fromindex) 返回位置
						 * 字符串值在字符串中首次出现的位置.第二个参数可选，从该fromindex位置（包括该位置）开始往后数
						 * var v = "abcabdefgh";
						 * alert(s.indexOf("ab", 1)); //弹出： 3
					 * s.lastIndexOf(searchvalue,fromindex)
						 * 字符串值最后出现的位置，在一个字符串中的指定位置从后向前搜索
				 * 截取字符串
					 * s.substring(start,stop)
						 * 负数无效，substring 会把所有负数转换为0,
						 start和stop参数谁大谁小都没关系，会自动更正
						 * 截取的元素不包括stop位置的元素
					 * s.slice(start,end) 与substring一样只是可以为负数，负数表示从末尾往前算，不过截取依然是往后截
						 * 注意，这里end的元素一定要在start位置的的元素之后？
					 * s.substr(start, length) 
						 * start可以为负数
					 * 注意以上三种方法，第二个参数都是可选，没指定的话默认到末尾。有指定stop时，都不包括stop位置的元素
						 * var s = "生命诚可贵";
						 * alert(s.substring(1));	// 弹出：命诚可贵
						 * alert(s.substring(1, 3)); // 弹出：命诚
						 * alert(s.slice(-3, -0))	// 弹出： 诚可
				 * 大小写转换 
					 * s.toLowerCase() 转小写
					 * s.toUpperCase() 转大写
				 * 去除字符串首尾空白字符
					 * s.trim() 
						var s = " \n \t ABc   Abc   \t \n \t";
						console.log(s)//\n是tab键。\ts是换行
						console.log(s.trim());//ABc   Abc
				 * 字符串替换方法
					 * s.replace(regexp/substr,replacement)
					 * 用一些字符替换另一些字符，或替换一个与正则表达式匹配的子串
					 * 第一个参数为字符串
					 * var s = "abcaba";
						// "ab"是普通的字符串，则只用js去替换第一个 ab
						var newStr = s.replace("ab", "js");	// jscaba
					 * 第一参数为正则表达式：
					 * 了解：[g]表示全局匹配，[i]表示不区分大小写
					 * alert(s.replace(/ab/gi, js));  //替换所有的ab为js，且忽略大小写
					 * 第二个参数为一个字符串值。规定了替换文本或生成替换文本的函数
					 * 
					 * s.match(匹配值)参数要么为字符串或正则表达式，返回的是数组
					 * var s = "abcaba";
					 * var arr = s.match("ab"); //因为是 普通字符串， 所以只匹配第一个
						alert(arr);	// ["ab"]
						alert(s.match(/ab/gi)); // ["ab", "ab"]
						//匹配所有数字
						num=timer.match（/\d+/gi)
			
						
					 */
					/**
					 * 第三种：s.search(匹配的参数) 
					 * 参数为正则表达式或字符串
					 * 返回的是第一个的下标而不是数组。不支持全局索引，没找到返回-1
					 */
					/**
				 * 字符串比较：
					 * 1==只比较数值，不比较类型
					 * 2=== 数值和类型都要比较
					 * 	对象字符串和基本数据类型字符串不一样的
					 * 就算都是对象，对象只跟自己相等，所以就算都是new的对象字符串s1和s2并且相同数值，也是不等的
					 * var s3 = new String("今天是是2016年11月5日");
						var s4 = new String("今天是是2016年11月5日");
						alert(s3 == s4);	//false？
						alert(s3 === s4);	//false
						//s3 和 s4 类型和内容都相等？，但是他们是对象，所以必须是同一个对象才算恒等。 面向对象阶段细讲
					 * 3s.localeCompare(other)
					 * 按ASCII进行比较，返回1（第一个排后面，）或0或-1，
					 * a与ab比，先比a，再比空和b，空=0=ascii:48,
					 * 中文比较国际编码表
					 */
					/**
				 * 字符串切割
					 * srt.split(separator,hanmany),返回数组
					 * 第一个参数为字符串或正则表达式，从该参数指定位置开始切割，比如用空格切，又如用指定字母切，切完后不包括该位置的字符
					 * 第二个为可选，指定返回的数组的长度，一般不设置
					 * var s = "How do you do";
						var arr = s.split(" ");	// 使用 " " 空格来切割字符串
						alert(arr.length);	// 4
						alert(arr);	// How,do,you,do
